data class Utilisateur(
    val nom: String,
    var email: String? // 'email' peut être null
)
fun afficherEmail(utilisateur: Utilisateur) {
    // Utilisation de l'opérateur Elvis (?:) pour fournir une valeur par défaut
    val emailAffiche = utilisateur.email ?: "Email non fourni"
    println("L'email de ${utilisateur.nom} est : $emailAffiche")
    // Utilisation de l'opérateur de chaînage sécurisé (?.)
    // Si 'utilisateur.email' est null, la partie droite de l'opérateur ne sera pas exécutée
    utilisateur.email?.let { emailNonNull ->
        println("Email sécurisé : $emailNonNull")
    }
    // Ou simplement pour un appel conditionnel :
    // utilisateur.email?.uppercase() // Retourne null si email est null
}
fun main() {
    val user1 = Utilisateur("Alice", "alice@example.com")
    val user2 = Utilisateur("Bob", null) // Bob n'a pas d'email
    afficherEmail(user1) // Affiche l'email réel
    afficherEmail(user2) // Affiche "Email non fourni" grâce à ?:
